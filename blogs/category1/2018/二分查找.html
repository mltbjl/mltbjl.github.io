    考小拉 7/18/2022 算法数组递归

    # 普通方法

    # 思路:

    ①首先需要得到一个有序的数组；

    ②定义数组的区间表示（左闭右闭区间 左闭右开区间）

    ③定义target值后，每次都要与中间值作对比。(以下图片出自up主的，借用来学习)

    （左闭右闭区间）如果 target > arr[mid] ，则 left = Mid + 1；target < arr[mid] ，则 right = Mid - 1；

    这边的while( left <= right ) left = right 有意义

    （左闭右开区间）如果 target > arr[mid] ，则 left = Mid；target < arr[mid] ，则 right = Mid；

    这边的while( left < right ) ，因为left = right无意义

    # 代码展示：

    1.区间为左闭右闭

    public int BinarySearch(int[] arr, int target) {
        int left = 0;
        int right = arr.length - 1;
        while (left <= right) { //注意这里这里的区间 是可以取 = 
            int mid = left + (right - left) / 2; //就是(left+right)/2
            if (target < arr[mid]) {
                right = mid - 1;
            } else if (target > arr[mid]) {
                left = mid + 1;
            } else {
                return mid;
            }
        }
        return -1;
    }

    public static int BinarySearch2(int[] arr, int target) {
        int left = 0;
        int right = arr.length - 1;
        while (left < right) {//注意这里这里的区间 是可以取 = ；因为右边的区间为开区间，取值没意义
            int mid = left + (right - left) / 2;
            if (target < arr[mid]) {
                right = mid;
            } else if (target > arr[mid]) {
                left = mid;
            } else {
                return mid;
            }
        }
        return -1;
    }

    # 二分查找（递归）

    # 代码展示：

    public static int BinarySearch3(int[] arr, int left, int right, int target) {
        if (left > right || (target < arr[0] || target > arr[arr.length - 1])) {
            return -1;
        }

        int mid = (left + right) / 2;
        if (target == arr[mid]) {
            return mid;
        } else if (target > arr[mid]) {
            return BinarySearch3(arr, mid + 1, right, target);
        } else {
            return BinarySearch3(arr, left, mid - 1, target);
        }
    }

    # 拓展（要查找的数有多个）

    # 代码展示：

     public ArrayList<Integer> BinarySearch(int[] arr, int target) {
        int left = 0;
        int right = arr.length - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2; //就是(left+right)/2
            if (target < arr[mid]) {
                right = mid - 1;
            } else if (target > arr[mid]) {
                left = mid + 1;
            } else {
                ArrayList<Integer> list = new ArrayList<Integer>();
                //扫描左边的元素
                int temp = mid - 1;
                while (true) {
                    if (temp < 0 || arr[temp] != target) {
                        break;
                    }
                    list.add(temp);
                    temp--;
                }
                list.add(mid);

                //扫描右边的元素
                temp = mid + 1;
                while (true) {
                    if (temp > arr.length - 1 || arr[temp] != target) {
                        break;
                    }
                    list.add(temp);
                    temp++;
                }
                return list;
            }
        }
        return new ArrayList<>();
    }
