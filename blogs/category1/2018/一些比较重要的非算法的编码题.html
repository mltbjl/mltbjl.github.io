    • 双重校验锁

      public class User {
    private static volatile User user;

    private User(){
    }

    /**
     * 这里有两个考点
     * 1. 为什么需要双重 null 的校验？
     * 比如开始有两个线程都进入了第一个null的代码块后，线程1拿到了锁，然后进行new 对象生成对象；释放锁后，线程2也拿到了锁，此时发现不为null了，则可以直接返回不需要再new对象
     * 2. 为什么需要加 volatile 关键字来修饰 user
     * ①可以会new两次这个对象，那么就不符合单例模式了；②禁止指令重排，这个 user =  new User()过程是 先给User分配一个内存，然后进行初始化，最后将这个内存指针指向这个内存；顺序是这样的如果不加volatile
     * 有可能颠倒顺序
     */
    public static User getInstance() {
        if (user == null) {
            synchronized (User.class) {
                if (user == null) {
                    user = new User();
                }
            }
        }
        return user;
    }
}

    • 抽象工厂模式

      抽象产品接口 Product 定义了产品的操作方法 operation。具体产品 ConcreteProductA 和 ConcreteProductB 实现了 Product 接口，并分别实现了自己的具体操作。

      抽象工厂接口 AbstractFactory 定义了创建产品的方法 createProduct。具体工厂 ConcreteFactoryA 和 ConcreteFactoryB 实现了 AbstractFactory 接口，分别创建 ConcreteProductA 和 ConcreteProductB。

      在 Main 类中，首先创建具体工厂A，并通过工厂A创建具体产品A，然后调用产品A的操作方法。接着创建具体工厂B，并通过工厂B创建具体产品B，最后调用产品B的操作方法。

      // 抽象产品接口
interface Product {
    void operation();
}

// 具体产品A
class ConcreteProductA implements Product {
    @Override
    public void operation() {
        System.out.println("具体产品A的操作");
    }
}

// 具体产品B
class ConcreteProductB implements Product {
    @Override
    public void operation() {
        System.out.println("具体产品B的操作");
    }
}

// 抽象工厂接口
interface AbstractFactory {
    Product createProduct();
}

// 具体工厂A
class ConcreteFactoryA implements AbstractFactory {
    @Override
    public Product createProduct() {
        return new ConcreteProductA();
    }
}

// 具体工厂B
class ConcreteFactoryB implements AbstractFactory {
    @Override
    public Product createProduct() {
        return new ConcreteProductB();
    }
}

public class Main {
    public static void main(String[] args) {
        // 创建具体工厂A
        AbstractFactory factoryA = new ConcreteFactoryA();
        // 通过工厂A创建具体产品A
        Product productA = factoryA.createProduct();
        productA.operation();

        // 创建具体工厂B
        AbstractFactory factoryB = new ConcreteFactoryB();
        // 通过工厂B创建具体产品B
        Product productB = factoryB.createProduct();
        productB.operation();
    }
}

    # 多线程编程

    多线程编程有一个 口诀：

    1. 线程操纵资源类
    2. 判断 --- 执行流程 --- 通知
    3. 判断虚假唤醒，用 while 不用 if
    4. 标志位（精准通知 condition）

    三角使用：synchronized --- wait --- notify && lock --- await --- condition

    # 两个线程，一个线程打印奇数，一个线程打印偶数

    • 资源类

      public class Source {
    private final Lock lock = new ReentrantLock(false);
    private final Condition conditionOdd = lock.newCondition();
    private final Condition conditionEven = lock.newCondition();
    int i = 1;

    /**
     * 打印奇数
     */
    public void printOdd() {
        lock.lock();
        try {
            // 阻塞我们的奇数线程
            while (i % 2 == 0) {
                try {
                    conditionOdd.await();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }

            System.out.println("这是奇数：" + i);
            i++;

            // 通知打印偶数的线程
            conditionEven.signal();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
    }

    /**
     * 打印偶数
     */
    public void printEven() {
        lock.lock();
        try {
            // 阻塞我们的偶数线程
            while (i % 2 != 0) {
                try {
                    conditionEven.await();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }

            System.out.println("这是偶数：" + i);
            i++;

            // 通知打印奇数的线程
            conditionOdd.signal();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
    }
}

    # 三个线程按序打印ABC

    • 资源类

      public class Main {
    public static void main(String[] args) {
        new Thread(Abc::printA, "A线程").start();
        new Thread(Abc::printB, "B线程").start();
        new Thread(Abc::printC, "C线程").start();
    }
}

class Abc {
    private static final ReentrantLock lock = new ReentrantLock(false);
    private static final Condition a = lock.newCondition();
    private static final Condition b = lock.newCondition();
    private static final Condition c = lock.newCondition();
    /**
     * 判断是否执行：1表示应该A执行，2表示应该B执行，3表示应该C执行
     */
    private static int state = 1;

    public static void printA() {
        while (state < 4) {
            // 上锁
            lock.lock();
            try {
                // 判断，避免虚假唤醒 用 while 不用 if
                while (state != 1) {
                    a.await();
                }
                // 执行流程
                System.out.println("A执行成功");
                state++;
                // 通知
                b.signal();
            } catch (Exception e) {
                e.printStackTrace();
            } finally {
                lock.unlock();
            }
        }
    }

    public static void printB() {
        while (state < 4) {
            // 上锁
            lock.lock();
            try {
                while (state != 2) {
                    b.await();
                }
                System.out.println("B执行成功");
                state++;
                c.signal();
            } catch (InterruptedException e) {
                throw new RuntimeException(e);
            } finally {
                lock.unlock();
            }
        }
    }

    public static void printC() {
        while (state < 4) {
            // 上锁
            lock.lock();
            try {
                while (state != 3) {
                    c.await();
                }
                System.out.println("C执行成功");
                state++;
            } catch (Exception e) {
                e.printStackTrace();
            } finally {
                lock.unlock();
            }
        }
    }
}

    # 三个线程，第一个线程输出 A 5次，紧接着第二个线程输出 B 10次，最后第三个线程输出 C 15次

    • 资源类

      public class Source {
    private final Lock lock = new ReentrantLock(false);
    private final Condition conditionA = lock.newCondition();
    private final Condition conditionB = lock.newCondition();
    private final Condition conditionC = lock.newCondition();
    /**
     * 1 表示A执行；2 表示B执行；3 表示C执行
     */
    private int state = 1;

    public void printA() throws InterruptedException {
        lock.lock();
        try {
            while (state != 1) {
                conditionA.await();
            }
            for (int i = 0; i < 5; i++) {
                System.out.println("A线程打印：" + (i + 1));
            }

            state = 2;
            // 通知线程B
            conditionB.signal();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
    }

    public void printB() throws InterruptedException {
        lock.lock();
        try {
            while (state != 2) {
                conditionB.await();
            }

            for (int i = 0; i < 10; i++) {
                System.out.println("B线程打印：" + (i + 1));
            }
            state = 3;
            conditionC.signal();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
    }

    public void printC() throws InterruptedException {
        lock.lock();
        try {
            while (state != 3) {
                conditionC.await();
            }

            for (int i = 0; i < 15; i++) {
                System.out.println("C线程打印：" + (i + 1));
            }
            state = 1;
            conditionA.signal();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
    }
}

    # 生产者消费者

    • 资源类

      public class Source {
    private Lock lock = new ReentrantLock(false);

    private Condition producer = lock.newCondition();

    private Condition consumer = lock.newCondition();
    /**
     * 1 代表生产者  2 代表消费者
     */
    private int state = 1;
    int i  = 1;

    public void put() {
        lock.lock();
        try {
            while (state != 1) {
                producer.await();
            }

            System.out.println("生产者生产" + (i++));
            state = 2;
            consumer.signal();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
    }

    public void get() {
        lock.lock();
        try {
            while (state != 2) {
                consumer.await();
            }

            System.out.println("消费者消费" + (--i));
            state = 1;
            producer.signal();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
    }
}

    # 快速排序

    class Solution {
    public static void main(String[] args) {
        int[] ints = new Solution().sortArray(new int[]{9, 25, 7, 678, 19, 14});
        System.out.println(Arrays.toString(ints));
    }

    public int[] sortArray(int[] nums) {
        quickSort(nums, 0, nums.length - 1);
        return nums;
    }

    /**
     * 第一种方式
     */
    private void quickSort(int[] nums, int left, int right) {
        //递归退出条件
        if (left >= right) {
            return;
        }
        int mid = left + (right - left) / 2;
        swap(nums, mid, right);
        // 基数
        int pivot = nums[right];
        int i = left,j = left;
        while (j <= right) {
            if (nums[j] <= pivot) {
                swap(nums, i, j);
                j++;
                i++;
            } else {
                j++;
            }
        }

        // 递归条件中 为什么是 i - 2 以及 i 呢？ 因为当 j 指针到达了 pivot 元素的时候，这时候 i 和 j 同时加加，而且又是左闭右闭区间，对于右区间来说，左区间的终点肯定是 i - 2的，而对于右区间来说，此时的 i 就是起点
        quickSort(nums, left, i - 2);
        quickSort(nums, i, right);
    }

    public void swap(int[] nums, int left, int right) {
        int tmp = nums[left];
        nums[left] = nums[right];
        nums[right] = tmp;
    }
}

    public int[] sortArray(int[] nums) {
        recursion(nums, 0, nums.length - 1);
        return nums;
    }

    public void recursion(int[] nums, int i, int j) {
        if (i >= j) {
            return;
        }
        int mid = i + j >> 1;
        // 左闭右闭
        recursion(nums, i, mid);
        recursion(nums, mid + 1, j);
        merge(nums, i, mid, j);
    }

    public void merge(int[] nums1, int left, int mid, int right) {
        int[] res = new int[right - left + 1];
        int i = left, j = mid + 1, k = 0;
        while (i <= mid && j <= right) {
            // 即使相等也不改变这个顺序，稳定性
            if (nums1[i] <= nums1[j]) {
                res[k++] = nums1[i++];
            } else if (nums1[i] > nums1[j]) {
                res[k++] = nums1[j++];
            }
        }
        while (i <= mid) {
            res[k++] = nums1[i++];
        }
        while (j <= right) {
            res[k++] = nums1[j++];
        }

        // 将res数组的数 转移在 nums1中
        for (int t = 0; t < res.length; ++t) {
            nums1[t + left] = res[t];
        }
    }
}

            return listNode2;

            return listNode1;

        ListNode dummyNode = new ListNode(-10001);
        ListNode pre = dummyNode;
        ListNode cur1 = listNode1;
        ListNode cur2 = listNode2;

        while (cur2 != null && cur1 != null) {
            if (cur1.val > cur2.val) {
                pre.next = cur2;
                cur2 = cur2.next;
            } else {
                pre.next = cur1;
                cur1 = cur1.next;

            cur1 = cur1.next;
            pre = pre.next;
        }
        while (cur2 != null) {
            pre.next = cur2;
            cur2 = cur2.next;
            pre = pre.next;

    # 冒泡排序

    每次都是两两对比，如果左边的数大于右边的那个数，那么就两两交换，直到索引最后一个数，最终达到一次交换；

    稍微优化的点：开始的时候加入一个标识符，如果是有序的那么就不需要再做无用功，直接退出循环即可。

    public class Solution {
    public int[] sort(int[] arr) {
        for (int i = len - 1; i >= 0; i--) {
            // 先默认数组是有序的，只要发生一次交换，就必须进行下一轮比较，
            // 如果在内层循环中，都没有执行一次交换操作，说明此时数组已经是升序数组
            boolean sorted = true;
            for (int j = 0; j < i; j++) {
                if (nums[j] > nums[j + 1]) {
                    swap(nums, j, j + 1);
                    sorted = false;
                }
            }
            if (sorted) {
                break;
            }
        }
        return nums;
    }
    public void swap(int[] arr,int left,int right) {
        int tmp = arr[left];
        arr[left] = arr[right];
        arr[right] = tmp;
    }
}

    # 插入排序

    第一轮遍历来说，暂存当前指针 i 指向的值 tmp，然后用另一个指针 j 进行遍历，如果 nums[j - 1] > tmp ；则进行覆盖操作：nums[j] = nums[j - 1] ，子循环中，不断用前面大的那个值覆盖后面的值，退出循环后，将当前 j 指向的那个值用 tmp 恢复即可。

    public class Solution {
    public int[] sort(int[] arr) {
        int len = nums.length;
        // 循环不变量：将 nums[i] 插入到区间 [0, i) 使之成为有序数组
        for (int i = 1;i < len;i++) {
            // 先暂存这个元素，然后之前的元素逐个后移，留出空位
            int tmp = nums[i];
            int j = i;
            // 注意边界 j > 0；nums[j - 1] > tmp 也有不符合条件的话提前终止排序交换的意思
            while (j > 0 && nums[j - 1] > tmp) {
                nums[j] = nums[j - 1];
                j--;
            }
            nums[j] = tmp;
        }
        return nums;
	}
}

    • 迭代

      分析一下为什么刚刚写的前序遍历的代码，不能和中序遍历通用呢，因为前序遍历的顺序是中左右，先访问的元素是中间节点，要处理的元素也是中间节点，所以刚刚才能写出相对简洁的代码，因为要访问的元素和要处理的元素顺序是一致的，都是中间节点。

      那么再看看中序遍历，中序遍历是左中右，先访问的是二叉树顶部的节点，然后一层一层向下访问，直到到达树左面的最底部，再开始处理节点（也就是在把节点的数值放进result数组中），这就造成了处理顺序和访问顺序是不一致的。

      那么在使用迭代法写中序遍历，就需要借用指针的遍历来帮助访问节点，栈则用来处理节点上的元素。

      // 前序遍历
class Main {
    public List<Integer> preOrderTraversal(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) {
            return result;
        }
        Stack<TreeNode> stack = new Stack<>();
        stack.push(root);
         while (!stack.isEmpty()){
            TreeNode node = stack.pop();
            result.add(node.val);
            if (node.right != null){
                stack.push(node.right);
            }
            if (node.left != null){
                stack.push(node.left);
            }
        }
        return result;
    }
}

// 中序遍历顺序: 左-中-右 入栈顺序： 左-右
class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null){
            return result;
        }
        
        Stack<TreeNode> stack = new Stack<>();
        TreeNode cur = root;
        while (cur != null || stack.isEmpty()) {
            if (cur != null) {
                stack.push(cur);
                cur = cur.left;
            } else {
                cur = stack.pop();
                result.add(cur.val);
                cur = cur.right;
            }
        }
        return result;
    }
}


// 后序遍历顺序 左-右-中 入栈顺序：中-左-右 出栈顺序：中-右-左， 最后翻转结果
class Solution {
    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null){
            return result;
        }
        Stack<TreeNode> stack = new Stack<>();
        stack.push(root);
        while (!stack.isEmpty()){
            TreeNode node = stack.pop();
            result.add(node.val);
            if (node.left != null){
                stack.push(node.left);
            }
            if (node.right != null){
                stack.push(node.right);
            }
        }
        Collections.reverse(result);
        return result;
    }
}

                if (tmpNode.right != null) que.offer(tmpNode.right);
            }

            resList.add(itemList);
        }

      public class Solution {
    public static void main(String[] args) {

        testBagProblemII(new int[]{1, 3, 4}, new int[]{15, 20, 30}, 4);

    public static void testWeightBagProblem(int[] weight, int[] value, int bagSize) {
        // dp数组的含义：从下标为[0-i]的物品里任意取，放进容量为j的背包 最大的价值为dp[i][j]
        int[][] dp = new int[weight.length][bagSize + 1];
        int length = weight.length;
        /* 初始化
         *  当背包为0的时候，任何东西都没办法装得下，价值dp[i][j] 自然都为0
         *  当背包的容量逐渐上升的时候，只要容量大于weight[0] 那么价值最大也就是value[0]
         * */
        for (int i = 0; i < length; i++) {
            dp[i][0] = 0;
        }
        for (int i = 0; i <= bagSize; i++) {
            if (i >= weight[0]) {
                dp[0][i] = value[0];
            }
        }

        // 确定遍历次序（先遍历物品，后遍历背包）
        for (int i = 1; i < length; i++) {
            for (int j = 1; j <= bagSize; j++) {
                if (weight[i] <= j) {
                    // 物品的重量 小于或者等于 背包的容量的时候
                    /*
                     不放物品dp[i - 1][j];
                     dp[i - 1][j - weight[i]] 为背包容量为j - weight[i]的时候不放物品i的最大价值
                     那么dp[i - 1][j - weight[i]] + value[i] 就是背包放物品i得到的最大价值
                     */
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i - 1][j - weight[i]] + value[i]);
                } else {
                    // 物品的重量 大于 背包的容量的时候
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
    }

        // 拿coins[0] = 1;coins[1] = 5 为例，背包的每个值都是经过1和5的计算，包含了{1, 5}和{5, 1}两种方式
        for (int i = 1; i <= amount; i++) {
            for (int j = 0; j < coins.length; j++) {
                if (i - coins[j] >= 0) {
                    dp[i] += dp[i - coins[j]];
                }
            }
        }*/


        return dp[amount];
    }
