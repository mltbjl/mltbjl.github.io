    # 构造方法

    # 作用

    • 创建对象
    • （多个属性同时）可以被初始化

    # 代码展示

    public class Students {
    private String name;

    /**
     * 类中默认是有无参构造方法的，但是一旦写了有参构造方法，不写无参构造方法，无参就会被覆盖
     * 类中肯定是要写无参构造方法的
     */

    //创建一个无参构造方法
    public Students(){
        
//        System.out.println();  //若写了这句就报错了
       // 构造方法相互调用 this(实参) 必须放在该方法体的第一行
//        this("lds");
         }

    //创建一个有参构造方法
    public Students(String name){
        
        //this 关键字表示的是类的对象
        this.name = name;
    }
}

    # 类的继承

    # 代码展示

    Students类

    public class Students {
    private String name;

    //创建一个无参构造方法
    public Students(){
//        System.out.println();  //若写了这句就报错了
        // 构造方法相互调用 this(实参) 必须放在该方法体的第一行
//        this("lds");
         }

    //创建一个有参构造方法
    public Students(String name){
        //this 关键字表示的是类的对象
        this.name = name;
    }

    public void setName(String name){
        this.name = name;
    }

    public String getName(){
        return name;
    }
    /**
     * 方法的重载和重写（一般用于继承的关系）
     * 重载：返回值、方法名要一样 形参 形参个数或者顺序 其中一个不一样就是方法重载
     * 重写：返回值、方法名、形参、形参个数 全都要一致 且访问修饰符要大于或者等于父类的
     */
     public void study(){
        System.out.println(name);
    }
}

    //子类会继承父类的属性和方法，但是私有的方法和属性不能直接调用
public class UNStudents extends Students{

    public UNStudents(){}

    public void printName(){
        System.out.println("名字："+getName());
    }

    public void study(String s){
        System.out.println(s+"在学习");
    }
    public void study(){
        /**
         * 保存父类的方法 关键字super 表示当前类的父类对象
         */
//        super.study();
        System.out.println("我在睡觉");
    }
}

    Test类

    public class Test {
    public static void main(String[] args) {
    
        Students students  =new UNStudents();
        /**
         * 这里说明
         * 虽然new的是UNStudents的，但是它的类型是父类的，且只扫描父类中有的方法
         * new一个子类的构造方法一定调用父类的构造方法
         */

        UNStudents unStudents = new UNStudents();

        /**
         * 自动转型以后只能调用子类对象中重写的方法
         * 强制转型(向下转型):前提是该对象先自动转型
         *  UNStudents unStudents =(UNStudents)students;
         */
        students.setName("lds");
        students.study();//这里输出的是父类的方法里的内容和子类的方法的内容
    }
}
